    "parameters": {
        "type": "object",
        "properties": {
            "location": {
                "type": "string",
                "description": "The city and state, e.g. San Francisco, CA",
            },

        },
        "required": ["location"],
    },
  }
]

  model="gpt-4-1106-preview",
  messages=messages,
  functions=functions,
